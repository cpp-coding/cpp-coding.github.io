模板系列-树的直径

概念与定义

我们将一棵树T=(V,E)T = ( V,E )T=(V,E)的直径定义为maxδ(u,v)(u,v∈V)maxδ ( u,v ) ( u,v ∈ V )maxδ(u,v)(u,v∈V)，也就是说，树中所有最短路径距离的最大值即为树的直径。

对于树的直径，主要有两种做法，一种是用两次 bfs(或者 dfs)，另一种是用树形 DP.

树形 DP 求解

对于每个节点我们要记录两个值：

f1[i]f1[i]f1[i]表示以iii为根的子树中,iii到叶子结点距离的最大值

f2[i]f2 [ i ]f2[i]表示以iii为根的子树中,iii到叶子结点距离的次大值

对于一个节点，它到叶子结点距离的最大值和次大致所经过的路径肯定是不一样的

若 j 是 i 的儿子，那么(下面的 w[i][j]w [ i ][ j ]w[i][j] 表示iii到jjj的路径长度)：

若 f1[i]<f1[j]+w[i][j]=>f2[i]=f1[i]f1 [ i ] < f1 [ j ] + w [ i ][ j ] => f2 [ i ] = f1 [ i ]f1[i]<f1[j]+w[i][j]=>f2[i]=f1[i],f1[i]=f1[j]+w[i][j]f1 [ i ] = f1 [ j ] + w [ i ][ j ]f1[i]=f1[j]+w[i][j];

否则, 若f2[i]<f1[j]+w[i][j]=>f2[i]=f1[j]+w[i][j]若 f2 [ i ] < f1 [ j ] + w [ i ][ j ]=>f2 [ i ] = f1 [ j ] + w [ i ][ j ]若f2[i]<f1[j]+w[i][j]=>f2[i]=f1[j]+w[i][j];

理解：这样做就是，先看能否更新最大值，若能，它的次大值就是原先的最大值，再更新它的最大值；若不能，就看能不能更新次大值，若能，就更新，不能就不管它

这样的话，最后的答案 answer=maxf1[i]+f2[i]answer = max { f1 [ i ] + f2 [ i ] }answer=maxf1[i]+f2[i]

#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=100005;
int n,m,t,ans;
int f1[N],f2[N];
int first[N],v[N],w[N],next[N];
void add(int x,int y,int z)
{
	t++;
	next[t]=first[x];
	first[x]=t;
	v[t]=y;
	w[t]=z;
}
void dp(int x,int father)
{
	int i,j;
	for(i=first[x];i;i=next[i])
	{
		j=v[i];
		if(j==father)
		  continue;
		dp(j,x);
		if(f1[x]<f1[j]+w[i])
		{
			f2[x]=f1[x];
			f1[x]=f1[j]+w[i];
		}
		else if(f2[x]<f1[j]+w[i])
		  f2[x]=f1[j]+w[i];
		ans=max(ans,f1[x]+f2[x]);
	}
}
int main()
{
	int x,y,z,i;
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;++i)
	{
		scanf("%d%d%d",&x,&y,&z);
		add(x,y,z);
		add(y,x,z);
	}
	dp(1,0);
	printf("%d",ans);
	return 0;
}
